                        import pandas as pd

def remove_outliers(df, column):
    """
    Função para remover outliers de uma coluna de um DataFrame
    usando o método do IQR (Interquartile Range).

    Parâmetros:
    df (pd.DataFrame): O DataFrame contendo os dados.
    column (str): O nome da coluna para verificar e remover os outliers.

    Retorna:
    pd.DataFrame: DataFrame com os outliers removidos.
    """
    
    # Calcular o primeiro quartil (Q1) e o terceiro quartil (Q3)
    Q1 = df[column].quantile(0.25)
    Q3 = df[column].quantile(0.75)
    
    # Calcular o intervalo interquartil (IQR)
    IQR = Q3 - Q1
    
    # Definir os limites inferior e superior para os outliers
    lower_limit = Q1 - 1.5 * IQR
    upper_limit = Q3 + 1.5 * IQR
    
    # Remover os outliers
    df_filtered = df[(df[column] >= lower_limit) & (df[column] <= upper_limit)]
    
    return df_filtered
